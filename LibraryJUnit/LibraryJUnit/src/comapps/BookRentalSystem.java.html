package comapps;

import java.util.ArrayList;

public class BookRentalSystem {
    
    private BookRentalSystem() {}
    
    public static ArrayList<Book> library = new ArrayList<>();

    // Adds books
    public static void addBooks(Book book) {
        library.add(book);
    }

    // Rents books based on index of library[]
    public static void rentBooks(int index) {
        if (index < library.size()) {
            library.get(index).rent();
        } else {
            System.out.println("Invalid index: " + index);
        }
    }

    // For resetting library[] when testing
    public static void clearLibrary() {
        library.clear();
    }

    // Getting size of library for test validation
    public static int getLibrarySize() {
        return library.size();
    }

    // Getter for library[]
    public static ArrayList<Book> getLibrary() {
        return library;
    }

    // Display all books
    public static void displayAllBooks() {
        for (Book b : library) {
            System.out.println(b.getTitle() + "\t" + b.getAuthor() + "\t" + b.getYearPublished());
        }
    }
    
    // Display only rented books
    public static void displayRentedBooks() {
        System.out.println("Books rented:");
        for (Book b : library) {
            if (b.isRented()) {
                System.out.println(b.getTitle() + "\t" + b.getAuthor() + "\t" + b.getYearPublished());
            }
        }
    }
    
    // Main method testing
    public static void main (String args[]) {
        // Add books
        BookRentalSystem.addBooks(new FictionBook("The Lord of the Rings", "J.R.R. Tolkien", 1954));
        BookRentalSystem.addBooks(new FictionBook("To Kill a Mockingbird", "Harper Lee", 1960));
        BookRentalSystem.addBooks(new NonFictionBook("The Tipping Point", "M. Gladwell", 2000));
        BookRentalSystem.addBooks(new NonFictionBook("Guns, Germs, and Steel", "Jared Diamond ", 1997));
        
        BookRentalSystem.rentBooks(0);
        BookRentalSystem.rentBooks(2);
        
        displayAllBooks();
        displayRentedBooks();
    }
}
